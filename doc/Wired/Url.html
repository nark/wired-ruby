def to_s
  	uri = URI::Generic.new(
	@scheme, 
	nil, 
	@hostname, 
	@port, 
	nil, nil, nil, nil, nil)

  	uri.user 		= @login
  	uri.password 	= @password
  	
  	return uri.to_s
end
